```js
// ── Pending meetings KPI card ─────────────────────────────────────────
async function loadPendingMeetings() {
  const qs = buildQs();  // passes group_id filter; backend ignores date/person params
  const data = await apiFetch(
    `/api/pending-meetings${qs}`,
    () => ({ count: 0, groups: [] })   // mock fallback: hide card
  );
  const card = document.getElementById('pending-meetings-card');
  if (!data.count || data.count === 0) {
    card.style.display = 'none';
    document.getElementById('pending-detail-panel').style.display = 'none';
    return;
  }
  card.style.display = '';   // restore to grid flow
  document.getElementById('stat-pending').textContent = data.count;
  window._pendingGroups = data.groups;  // stored for detail render
}

function renderPendingDetail() {
  const groups = window._pendingGroups || [];
  const content = document.getElementById('pending-detail-content');
  if (!groups.length) { content.innerHTML = ''; return; }
  content.innerHTML = `<ul class="alert-card__list">
    ${groups.map(g => `
      <li class="alert-card__item" style="flex-direction:column;align-items:flex-start;gap:4px;padding:12px 16px">
        <span class="alert-card__name">${g.name || g.id}</span>
        <span class="alert-card__meta" style="font-size:11px;color:var(--muted)">${g.missing_weeks.join(' · ')}</span>
      </li>`).join('')}
  </ul>`;
}

// ── Qualification panel ───────────────────────────────────────────────
```
